        // Simulación lógica de validación de acceso

        function ejecutarLogin() {

            const user = document.getElementById('usuario').value;

            const pass = document.getElementById('clave').value;

            if(user.trim() !== "" && pass.trim() !== "") {

                document.getElementById('pantalla-login').style.display = 'none';

                document.getElementById('dashboard').style.display = 'flex';

                mostrarToast('Acceso Especialista Autorizado');

            } else {

                alert('Por favor introduzca sus credenciales de especialista.');

            }

        }

        function loginBiometrico() {

            // Bypass estético directo para APK Creator

            document.getElementById('pantalla-login').style.display = 'none';

            document.getElementById('dashboard').style.display = 'flex';

            mostrarToast('Acceso Biométrico Correcto');

        }

        // Lógica de Mensajería Instantánea Local

        function enviarMensajeChat() {

            const input = document.getElementById('chat-mensaje');

            const texto = input.value.trim();

            if (texto === "") return;

            const cuerpo = document.getElementById('chat-cuerpo');

           

            // Render del mensaje enviado por ti

            const msgEnviado = document.createElement('div');

            msgEnviado.className = 'mensaje enviado';

            msgEnviado.innerText = texto;

            cuerpo.appendChild(msgEnviado);

            input.value = "";

            cuerpo.scrollTop = cuerpo.scrollHeight;

            // Simulación de respuesta inmediata de un colega para validar el flujo

            setTimeout(() => {

                const msgRecibido = document.createElement('div');

                msgRecibido.className = 'mensaje recibido';

                msgRecibido.innerText = "Entendido. Revisando los puntos en la plataforma.";

                cuerpo.appendChild(msgRecibido);

                cuerpo.scrollTop = cuerpo.scrollHeight;

                lanzarNotificacion('Nuevo mensaje en Konecta', 'Un colega ha respondido en el canal interno.');

            }, 1500);

        }

        // Persistencia de Informes de reunión

        function guardarInformeMinuta() {

            const textoMinuta = document.getElementById('campo-minuta').value;

            localStorage.setItem('konecta_minuta_data', textoMinuta);

            mostrarToast('Informe guardado de forma segura.');

        }

        // Sistema de Alertas Visuales (Toasts)

        function mostrarToast(mensaje) {

            const toast = document.getElementById('toast');

            toast.innerText = mensaje;

            toast.className = 'mostrar';

            setTimeout(() => { toast.className = ''; }, 3000);

        }

        // Integración con el Sistema de Notificaciones del dispositivo Android

        function lanzarNotificacion(titulo, info) {

            if ('Notification' in window && Notification.permission === 'granted') {

                new Notification(titulo, { body: info });

            } else if ('Notification' in window && Notification.permission !== 'denied') {

                Notification.requestPermission().then(permission => {

                    if (permission === 'granted') new Notification(titulo, { body: info });

                });

            }

        }
